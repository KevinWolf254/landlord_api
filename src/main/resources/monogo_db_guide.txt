1. check status 
sudo systemctl status mongod.service

1.1 if not running restart service
sudo systemctl restart mongod.service

2. Check version, address and port
mongo --eval 'db.runCommand({ connectionStatus: 1 })'

3. start using mongodb
mongo

4. show databases
show dbs

5. use admin inorder to create username and password
use admin

6. create user
db.createUser(
  {
    user: "kevin",
    pwd: "admin",
    roles: [ { role: "userAdminAnyDatabase", db: "admin" }, "readWriteAnyDatabase" ]
  }
)

7. cd to /etc
8. change mongod.conf file using vi editor
sudo vi mongod.conf

9. enable security and save file
security:
	authorization: enabled

10. restart mongod
sudo systemctl restart mongod.service

11. connect to mongo using username and password
mongo -u "kevin" -p "admin" --authenticationDatabase "admin"
